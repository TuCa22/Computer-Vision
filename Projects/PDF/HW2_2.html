def compute_hist(img):
    hist = np.zeros((256,), np.uint32)
    h, w = img.shape[:2]
    for i in range(h):
        for j in range(w):
            hist[img[i][j]] += 1
    return hist

def equal_hist(img):
    hist = np.zeros((256,), np.uint32)
    [w, h] = img.shape[:2]
    for i in range(w):
        for j in range(h):
            hist[img[i][j]]+=1
    cdf = np.cumsum(hist)
    sk = np.uint32(256 * cdf / np.max(cdf))
    return sk[img]

def gamma_ajust(image, gamma):
  image = np.power(image, gamma)
  max_val = np.max(image.ravel())
  image = image/max_val * 255
  image = image.astype(np.uint8)
  return image

# Thiết lập các điểm của biến đổi tuyến tính theo mảnh
points_in = np.array([[0, 0], [50, 80], [150, 100], [255, 255]], dtype=np.uint8)
points_out = np.array([[0, 0], [100, 50], [150, 200], [255, 255]], dtype=np.uint8)

# Tạo hàm ánh xạ từ các điểm đã cho
trans_func = np.zeros((256,), dtype=np.uint8)

# Xác định hàm ánh xạ
for i in range(len(points_in) - 1):
    x1, y1 = points_in[i]
    x2, y2 = points_in[i + 1]
    for x in range(x1, x2 + 1):
        y = int(y1 + (x - x1) * (y2 - y1) / (x2 - x1))
        trans_func[x] = np.clip(y, 0, 255)

# Biến đổi ảnh sử dụng hàm ánh xạ
PLT_img_1 = cv2.LUT(img_gray_1, trans_func)
PLT_img_2 = cv2.LUT(img_gray_2, trans_func)
PLT_img_3 = cv2.LUT(img_gray_3, trans_func)

def local_hist_equal(image, clip_limit=20, grid_size=(8, 8)):
    clahe = cv2.createCLAHE(clipLimit=clip_limit, tileGridSize=grid_size)
    equalized_image = clahe.apply(image)
    return equalized_image
